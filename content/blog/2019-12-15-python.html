def mers(sequence): hold = {}

    for i in range(len(sequence) - 1):
        if sequence[i:i + 3] in hold:
            hold[sequence[i:i +3]] +=1
        else:
            hold[sequence[i:i + 3]] = 1
    return hold

print(mers(my_seq1)) print(mers(my_seq2))
